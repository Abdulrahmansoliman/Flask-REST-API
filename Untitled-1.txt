#include <iostream>
#include <algorithm>
#include <vector>
#include <bits/stdc++.h>
    using namespace std;
    int main()
    {
        int t; cin >> t;
        while (t--)
        {
            int ref = 0, max_ram = 0;
            int n, k; cin >> n >> k;
            vector <int> v1;
            vector <int> v2;
            for(int i=0;i<n;i++)
            {
                int x; cin >> x;
                v1.push_back(x);
            }
            for (int i = 0; i < n; i++)
            {
                int y; cin >> y;
                v2.push_back(y);
            }
            vector <pair<int, int>> v;
            for (int i = 0; i < n; i++)
                v.push_back({ v1[i],v2[i] });
            for (int i = 0; i < n; i++)
            {
                if (v[i].first <= k)
                {
                    ref = i;
                    max_ram = v[i].first + v[i].second;
                }
                
            }
            if (max_ram == 0)
            {
                cout << k;
                return 0;
            }
            v.erase(v.begin() + ref);
            sort(v.begin(),v.end());
            
            for (int i = 0; i < n - 1; i++)
            {
                if (v[i].first <= max_ram)
                {
                    max_ram += v[i].second;
                }
            }
            cout << max_ram << '\n';
       
    }
        
    }
